\section{
  \title{Appendix A: Connecting}

  \section[title=PostgreSQL]{
    \b{Required}

    \deflist{
      \term{\c{:name}}
      \def{Database name.}

      \term{\c{:user}}
      \def{User name.}

      \term{\c{:pass}}
      \def{User password.}
    }

    \b{Optional}

    \deflist{

      \term{\c{:host}}
      \def{Host that runs the database server. Default: \c{localhost}.}

      \term{\c{:port}}
      \def{Port the database server listens on. Default: 5432.}

      \term{\c{:ssl}}

      \def{\c{:yes} enables secure SSL connections to the server. This might be
        useful if the server is running on a host other than the default. Note
        that OpenSSL must be installed on both machines. For mpre information,
        see the relevant
        \link[uri=http://www.postgresql.org/docs/9.1/static/ssl-tcp.html]{PostgreSQL
          documentation}. Default: \c{:no}.}
    }
  }
}
